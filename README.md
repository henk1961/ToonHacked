# Use your rooted Toon Using HomeBridge

add this to your config.json  
"accessories": [
        {
            "accessory": "Thermostat",
            "name": "Toon Thermostaat",
            "apiroute": "http://yourToonIp",
            //optional
            "maxTemp": "26",
            "minTemp": "15",
        }
